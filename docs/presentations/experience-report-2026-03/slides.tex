\begin{frame}[label={sec:org6bdf072},fragile]{Emacs All The Way Down: An Experience Report}
 An MCP server that exposes the full Emacs \texttt{obarray} as tools.

\begin{itemize}
\item Started: Python shim wrapping \texttt{emacsclient}
\item Ended: Pure Elisp MCP server --- \textasciitilde{}280 lines, no external deps
\item Duration: One session (2026-03-17 to 2026-03-18)
\item 27+ agents, 36 commits, 11 conjectures
\end{itemize}
\end{frame}
\begin{frame}[label={sec:org8bc7a73},fragile]{The Axiom}
 \begin{quote}
The MCP server does not know what Emacs can do. Emacs tells it.
\end{quote}
\begin{block}{All tool definitions derived at runtime from \texttt{obarray} introspection}
\end{block}
\begin{block}{No function list is hardcoded}
\end{block}
\begin{block}{If you are hardcoding function names, you have violated the axiom}
\end{block}
\end{frame}

\begin{frame}[label={sec:org1e1587e},fragile]{C-011: Stdout Buffering --- The Real Bug}
 \texttt{princ} + \texttt{terpri} in Emacs batch mode does not flush stdout to a pipe.
\begin{block}{4 agents investigated, 6 approaches tested}
\begin{center}
\begin{tabular}{lll}
Approach & Works? & Notes\\
\hline
\texttt{princ} + \texttt{terpri} & NO & libc stdio fully buffered on pipes\\
\texttt{princ} + \texttt{sit-for 0} & NO & sit-for does not trigger fflush\\
\texttt{send-string-to-terminal} & YES & Direct fd 1 write, cross-platform\\
\texttt{write-region /dev/stdout} & YES & Bypasses libc via write(2), Unix only\\
\texttt{write-region /dev/fd/1} & YES & Same mechanism, macOS/Linux\\
\texttt{call-process-region} + \texttt{cat} & NO & Output goes to standard-output var\\
\end{tabular}
\end{center}
\end{block}
\begin{block}{\texttt{send-string-to-terminal} chosen --- writes directly to fd 1}
\end{block}
\begin{block}{Secondary fix: daemon check hanging on unreachable remote sockets}
Details: \href{../../conjectures/c-011-stdout-buffering.md}{docs/conjectures/c-011-stdout-buffering.md}
\end{block}
\end{frame}

\begin{frame}[label={sec:org4418020},fragile]{C-006 Deep Dive}
 Vanilla (-Q) = 8460 functions. Configured = 10040. Delta = 1580 (18.7\%).
\begin{block}{5 hypotheses, all confirmed}
\begin{center}
\begin{tabular}{ll}
Finding & Detail\\
\hline
\texttt{-{}-{}batch} implies \texttt{-q} & No user init loaded in batch mode\\
Filter masks 93\% of delta & Heuristic reduces 1580 to \textasciitilde{}110 extra\\
222 hidden autoloads & \texttt{help-function-arglist} returns sentinel\\
C primitives identical (1336) & init.el adds only Elisp-defined fns\\
Category error in test & Vanilla vs configured test was comparing wrong scope\\
\end{tabular}
\end{center}
\end{block}
\begin{block}{The text-consumer heuristic acts as a natural stabilizer}
\end{block}
\begin{block}{Core tool set is stable regardless of user configuration}
\end{block}

\end{block}

\begin{frame}[label={sec:org60ae4ce},fragile]{C-009: The Autoload Barrier}
 222 text-consumer functions hidden by unavailable arglists.
\begin{block}{Problem: \texttt{help-function-arglist} returns sentinel for unloaded autoloads}
\end{block}
\begin{block}{Resolving would yield 28.8\% increase in tool count}
5 approaches under investigation:

\begin{center}
\begin{tabular}{lll}
Approach & Mechanism & Risk\\
\hline
A: Selective autoload-do-load & Force-load matching symbols & Side effects on load\\
B: Parse autoload objects & Extract arglist from plist & Rarely present\\
C: Parse .elc headers & Read bytecode without eval & Fragile, circular dep\\
D: Bulk-load loaddefs.el & Replay normal startup & Violates -Q constraint\\
E: Static arglist map & Grep .el source at build time & Stale if Emacs updates\\
\end{tabular}
\end{center}

\begin{frame}[label={sec:orgbe1f949},fragile]{What We Discovered}
 \begin{block}{Batch mode stdout buffering (C-011)}
\begin{block}{\texttt{princ} does not flush to pipes --- protocol-breaking for interactive MCP}
\end{block}
\begin{block}{\texttt{send-string-to-terminal} is the cross-platform fix}
\end{block}
\end{block}
\begin{block}{Daemon instability under concurrent agents}
\begin{block}{3 daemon PIDs in 2 minutes with 9 agents}
\end{block}
\begin{block}{No coordination protocol for daemon lifecycle}
\end{block}
\end{block}
\begin{block}{Text property leaks}
\begin{block}{\texttt{json-serialize} returns unibyte strings in batch mode}
\end{block}
\begin{block}{Requires \texttt{decode-coding-string} before output}
\end{block}
\end{block}
\begin{block}{Arglist heuristic is imprecise (C-002)}
\begin{block}{40\% precision vs. 80\% conjectured}
\end{block}
\begin{block}{\texttt{object=/=buffer} args dominate false positives}
\end{block}
\end{block}
